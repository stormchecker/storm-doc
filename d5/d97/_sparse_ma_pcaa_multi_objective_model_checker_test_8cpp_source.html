256template<typename ValueType>
257bool expectPointContained(std::vector<std::vector<ValueType>> const& pointset, std::vector<ValueType> const& point, ValueType precision) {
258 for (auto const& p : pointset) {
259 EXPECT_EQ(p.size(), point.size()) << "Missmatch in point dimension.";
260 bool found = true;
261 for (uint64_t i = 0; i < p.size(); ++i) {
262 if (storm::utility::abs<ValueType>(p[i] - point[i]) > precision) {
263 found = false;
264 break;
265 }
266 }
267 if (found) {
268 return true;
269 }
270 }
271 // prepare failure message:
272 std::stringstream errstr;
273 errstr << "Point [";
274 bool firstPi = true;
275 for (auto const& pi : point) {
276 if (firstPi) {
277 firstPi = false;
278 } else {
279 errstr << ", ";
280 }
281 errstr << pi;
282 }
283 errstr << "] is not contained in point set {";
284 bool firstP = true;
285 for (auto const& p : pointset) {
286 if (firstP) {
287 firstP = false;
288 } else {
289 errstr << ", \t";
290 }
291 errstr << "[";
292 firstPi = true;
293 for (auto const& pi : p) {
294 if (firstPi) {
295 firstPi = false;
296 } else {
297 errstr << ", ";
298 }
299 errstr << pi;
300 }
301 errstr << "]";
302 }
303 errstr << "}.";
304 ADD_FAILURE() << errstr.str();
305 return false;
306}

308template<typename ValueType>
309bool expectSubset(std::vector<std::vector<ValueType>> const& lhs, std::vector<std::vector<ValueType>> const& rhs, ValueType precision) {
310 for (auto const& p : lhs) {
311 if (!expectPointContained(rhs, p, precision)) {
312 return false;
313 }
314 }
315 return true;
316}
317
318template<typename ValueType>
319std::vector<std::vector<ValueType>> convertPointset(std::vector<std::vector<std::string>> const& in) {
320 std::vector<std::vector<ValueType>> out;
321 for (auto const& point_str : in) {
322 out.emplace_back();
323 for (auto const& pi_str : point_str) {
324 out.back().push_back(storm::utility::convertNumber<ValueType>(pi_str));
325 }
326 }
327 return out;
328}
