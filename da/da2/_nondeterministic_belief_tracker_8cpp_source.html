15 numberActionsPerObservation = std::vector<uint64_t>(pomdp.getNrObservations(), 0);

175template<typename ValueType>
176void SparseBeliefState<ValueType>::updateHelper(std::vector<std::map<uint64_t, ValueType>> const& partialBeliefs, std::vector<ValueType> const& sums,
177 typename std::map<uint64_t, ValueType>::const_iterator nextStateIt, uint32_t newObservation,
178 std::unordered_set<SparseBeliefState<ValueType>>& previousBeliefs) const {
179 if (nextStateIt == belief.end()) {
180 for (uint64_t i = 0; i < partialBeliefs.size(); ++i) {
181 auto const& partialBelief = partialBeliefs[i];
182 auto const& sum = sums[i];
183 if (storm::utility::isZero(sum)) {
184 continue;
185 }
186 std::size_t newHash = 0;
187 ValueType risk = storm::utility::zero<ValueType>();
188 std::map<uint64_t, ValueType> finalBelief;
189 for (auto& entry : partialBelief) {
190 assert(!storm::utility::isZero(sum));
191 finalBelief[entry.first] = entry.second / sum;
192 // boost::hash_combine(newHash, std::hash<ValueType>()(entry.second));
193 boost::hash_combine(newHash, entry.first);
194 risk += entry.second / sum * manager->getRisk(entry.first);
195 }
196 previousBeliefs.insert(SparseBeliefState<ValueType>(manager, finalBelief, newHash, risk, id));
197 }
198 } else {
199 uint64_t state = nextStateIt->first;
200 auto newNextStateIt = nextStateIt;
201 newNextStateIt++;
202 std::vector<std::map<uint64_t, ValueType>> newPartialBeliefs;
203 std::vector<ValueType> newSums;
204 for (uint64_t i = 0; i < partialBeliefs.size(); ++i) {
205 for (auto row = manager->getPomdp().getNondeterministicChoiceIndices()[state];
206 row < manager->getPomdp().getNondeterministicChoiceIndices()[state + 1]; ++row) {
207 std::map<uint64_t, ValueType> newPartialBelief = partialBeliefs[i];
208 ValueType newSum = sums[i];
209 for (auto const& transition : manager->getPomdp().getTransitionMatrix().getRow(row)) {
210 if (newObservation != manager->getPomdp().getObservation(transition.getColumn())) {
211 continue;
212 }
213
214 if (newPartialBelief.count(transition.getColumn()) == 0) {
215 newPartialBelief[transition.getColumn()] = transition.getValue() * nextStateIt->second;
216 } else {
217 newPartialBelief[transition.getColumn()] += transition.getValue() * nextStateIt->second;
218 }
219 newSum += transition.getValue() * nextStateIt->second;
220 }
221 newPartialBeliefs.push_back(newPartialBelief);
222 newSums.push_back(newSum);
223 }
224 }
225 updateHelper(newPartialBeliefs, newSums, newNextStateIt, newObservation, previousBeliefs);
226 }

237bool NondeterministicBeliefTracker<ValueType, BeliefState>::reset(uint32_t observation) {
238 beliefs = std::unordered_set<BeliefState>();
239 bool hit = false;
240 for (auto state : pomdp.getInitialStates()) {
241 if (observation == pomdp.getObservation(state)) {
242 hit = true;
243 beliefs.emplace(manager, state);
244 }
245 }
246 lastObservation = observation;
247 return hit;

250template<typename ValueType, typename BeliefState>
251bool NondeterministicBeliefTracker<ValueType, BeliefState>::track(uint64_t newObservation) {
252 STORM_LOG_THROW(!beliefs.empty(), storm::exceptions::InvalidOperationException, "Cannot track without a belief (need to reset).");
253 std::unordered_set<BeliefState> newBeliefs;
254 storm::utility::Stopwatch trackTimer(true);
255 for (auto const& belief : beliefs) {
256 belief.update(newObservation, newBeliefs);
257 if (options.trackTimeOut > 0 && static_cast<uint64_t>(trackTimer.getTimeInMilliseconds()) > options.trackTimeOut) {
258 return false;
259 }
260 }
261 beliefs = newBeliefs;
262 lastObservation = newObservation;
263 return !beliefs.empty();
264}

268 STORM_LOG_THROW(!beliefs.empty(), storm::exceptions::InvalidOperationException, "Risk is only defined for beliefs (run reset() first).");

270 if (max) {
271 for (auto const& belief : beliefs) {
272 if (belief.getRisk() > result) {
273 result = belief.getRisk();

278 if (belief.getRisk() < result) {
279 result = belief.getRisk();

281 }
282 }

316template<typename ValueType, typename BeliefState>
317uint64_t NondeterministicBeliefTracker<ValueType, BeliefState>::reduce() {
318 reductionTimedOut = false;
319 std::shared_ptr<storm::utility::solver::SmtSolverFactory> solverFactory = std::make_shared<storm::utility::solver::Z3SmtSolverFactory>();
320 storm::storage::geometry::ReduceVertexCloud<ValueType> rvc(solverFactory, options.wiggle, options.timeOut);
321 std::vector<std::map<uint64_t, ValueType>> points;
322 std::vector<typename std::unordered_set<BeliefState>::iterator> iterators;
323 for (auto it = beliefs.begin(); it != beliefs.end(); ++it) {
324 // TODO get rid of the getBeliefMap function.
325 points.push_back(it->getBeliefMap());
326 iterators.push_back(it);
327 }
328 auto res = rvc.eliminate(points, pomdp.getNumberOfStates());
329 storm::storage::BitVector eliminate = ~res.first;
330 if (res.second) {
331 reductionTimedOut = true;
332 }
333
334 auto selectedIterators = storm::utility::vector::filterVector(iterators, eliminate);
335 for (auto iter : selectedIterators) {
336 beliefs.erase(iter);
337 }
338 return eliminate.getNumberOfSetBits();
339}
