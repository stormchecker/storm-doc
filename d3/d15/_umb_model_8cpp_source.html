104void UmbModel::encodeRationals() {
105 auto getSize = [](storm::umb::GenericVector& v) -> uint64_t {
106 if (v.isType<storm::RationalNumber>()) {
107 return ValueEncoding::getMinimalRationalSize(v.template get<storm::RationalNumber>(), true);
108 }
109 return 0ull;
110 };
111 auto encodeForSize = [](storm::umb::GenericVector& v, uint64_t const size) {
112 if (v.isType<storm::RationalNumber>()) {
113 auto values = ValueEncoding::createUint64FromRationalRange(v.template get<storm::RationalNumber>(), size);
114 v.template set<uint64_t>(std::move(values));
115 }
116 };
117 auto encode = [&getSize, &encodeForSize](storm::umb::GenericVector& v, std::optional<uint64_t>& size) {
118 if (v.isType<storm::RationalNumber>()) {
119 size = getSize(v);
120 encodeForSize(v, size.value());
121 }
122 };
123
124 if (index.transitionSystem.branchProbabilityType.has_value()) {
125 encode(branchToProbability, index.transitionSystem.branchProbabilityType->size);
126 }
127 if (index.transitionSystem.exitRateType.has_value()) {
128 encode(stateToExitRate, index.transitionSystem.exitRateType->size);
129 }
130 if (index.transitionSystem.observationProbabilityType.has_value()) {
131 if (stateObservations.has_value()) {
132 encode(stateObservations->probabilities, index.transitionSystem.observationProbabilityType->size);
133 }
134 if (branchObservations.has_value()) {
135 encode(branchObservations->probabilities, index.transitionSystem.observationProbabilityType->size);
136 }
137 }
138 for (auto& [annotationType, annotation] : annotations) {
139 for (auto& [annotationName, annotationValues] : annotation) {
140 // Get encoding bit sizes. Note that if the annotation applies to multiple entities, we need to use the maximum size.
141 uint64_t valueSize{0}, probSize{0};
142 boost::pfr::for_each_field(annotationValues, [&valueSize, &probSize, &getSize](auto& entity) {
143 if (entity.has_value()) {
144 valueSize = std::max(valueSize, getSize(entity->values));
145 probSize = std::max(probSize, getSize(entity->probabilities));
146 }
147 });
148 // encode
149 if (valueSize > 0 || probSize > 0) {
150 STORM_LOG_ASSERT(index.annotation(annotationType).has_value() && index.annotation(annotationType)->contains(annotationName),
151 "There are annotation files stored for annotations/" << annotationType << "/" << annotationName
152 << ", but no corresponding entry in the index.");
153 auto& indexAnnotation = index.annotation(annotationType)->at(annotationName);
154 if (valueSize > 0) {
155 indexAnnotation.type.size = valueSize;
156 boost::pfr::for_each_field(annotationValues, [&valueSize, &encodeForSize](auto& entity) {
157 if (entity.has_value()) {
158 encodeForSize(entity->values, valueSize);
159 }
160 });
161 }
162 if (probSize > 0) {
163 STORM_LOG_ASSERT(indexAnnotation.probabilityType.has_value(), "Found annotation probabilities but no type in index.");
164 indexAnnotation.probabilityType.value().size = probSize;
165 boost::pfr::for_each_field(annotationValues, [&probSize, &encodeForSize](auto& entity) {
166 if (entity.has_value()) {
167 encodeForSize(entity->probabilities, probSize);
168 }
169 });
170 }
171 }
172 }
173 }

176void UmbModel::decodeRationals() {
177 auto decode = [](storm::umb::GenericVector& v, std::optional<SizedType> const type) {
178 if (type.has_value() && type->type == Type::Rational && v.isType<uint64_t>()) {
179 auto valueView = ValueEncoding::uint64ToRationalRangeView(v.template get<uint64_t>(), type->bitSize());
180 std::vector<storm::RationalNumber> values(valueView.begin(), valueView.end());
181 v.template set<storm::RationalNumber>(std::move(values));
182 }
183 };
184
185 if (index.transitionSystem.branchProbabilityType.has_value()) {
186 decode(branchToProbability, index.transitionSystem.branchProbabilityType.value());
187 }
188 if (index.transitionSystem.exitRateType.has_value()) {
189 decode(stateToExitRate, index.transitionSystem.exitRateType.value());
190 }
191 if (index.transitionSystem.observationProbabilityType.has_value()) {
192 if (stateObservations.has_value()) {
193 decode(stateObservations->probabilities, index.transitionSystem.observationProbabilityType.value());
194 }
195 if (branchObservations.has_value()) {
196 decode(branchObservations->probabilities, index.transitionSystem.observationProbabilityType.value());
197 }
198 }
199
200 for (auto& [annotationType, annotation] : annotations) {
201 for (auto& [annotationName, annotationValues] : annotation) {
202 STORM_LOG_ASSERT(index.annotation(annotationType).has_value() && index.annotation(annotationType)->contains(annotationName),
203 "There are annotation files stored for annotations/" << annotationType << "/" << annotationName
204 << ", but no corresponding entry in the index.");
205 auto const& annotationIndex = index.annotation(annotationType)->at(annotationName);
206 boost::pfr::for_each_field(annotationValues, [&annotationIndex, &decode](auto& entity) {
207 if (entity.has_value()) {
208 decode(entity->values, annotationIndex.type);
209 decode(entity->probabilities, annotationIndex.probabilityType.value());
210 }
211 });
212 }
213 }
