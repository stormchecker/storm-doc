98storm::solver::NativeLinearEquationSolverMethod NativeEquationSolverSettings::getLinearEquationSystemMethod() const {
99 std::string linearEquationSystemTechniqueAsString = this->getOption(techniqueOptionName).getArgumentByName("name").getValueAsString();
100 if (linearEquationSystemTechniqueAsString == "jacobi") {
101 return storm::solver::NativeLinearEquationSolverMethod::Jacobi;
102 } else if (linearEquationSystemTechniqueAsString == "gaussseidel") {
103 return storm::solver::NativeLinearEquationSolverMethod::GaussSeidel;
104 } else if (linearEquationSystemTechniqueAsString == "sor") {
105 return storm::solver::NativeLinearEquationSolverMethod::SOR;
106 } else if (linearEquationSystemTechniqueAsString == "walkerchae") {
107 return storm::solver::NativeLinearEquationSolverMethod::WalkerChae;
108 } else if (linearEquationSystemTechniqueAsString == "power") {
109 return storm::solver::NativeLinearEquationSolverMethod::Power;
110 } else if (linearEquationSystemTechniqueAsString == "sound-value-iteration" || linearEquationSystemTechniqueAsString == "svi") {
111 return storm::solver::NativeLinearEquationSolverMethod::SoundValueIteration;
112 } else if (linearEquationSystemTechniqueAsString == "optimistic-value-iteration" || linearEquationSystemTechniqueAsString == "ovi") {
113 return storm::solver::NativeLinearEquationSolverMethod::OptimisticValueIteration;
114 } else if (linearEquationSystemTechniqueAsString == "guessing-value-iteration" || linearEquationSystemTechniqueAsString == "gvi") {
115 return storm::solver::NativeLinearEquationSolverMethod::GuessingValueIteration;
116 } else if (linearEquationSystemTechniqueAsString == "interval-iteration" || linearEquationSystemTechniqueAsString == "ii") {
117 return storm::solver::NativeLinearEquationSolverMethod::IntervalIteration;
118 } else if (linearEquationSystemTechniqueAsString == "ratsearch") {
119 return storm::solver::NativeLinearEquationSolverMethod::RationalSearch;
120 }
121 STORM_LOG_THROW(false, storm::exceptions::IllegalArgumentValueException,
122 "Unknown solution technique '" << linearEquationSystemTechniqueAsString << "' selected.");
123}
